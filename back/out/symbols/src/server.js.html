119 function loadFile(date) {
120     lines = []
121     fs.createReadStream(`${baseFolder}/${date}/index.txt`).on('error', (err) => {
122         res.status(404).json(err)
123     })
124         .pipe(es.split())
125         .pipe(es.map(function (line, next) {
126             var tokens = line.split('/')
127             lines.push({ folder: tokens.slice(0, -1).join("/"), file: tokens[tokens.length - 1] })
128             next(null, line);
129         })).pipe(es.wait(function (err, body) {
130             console.log(lines.length);
131             // decompose "20250829_013916_049_48_msgin_BatchAAAFlow_86caba26-278d-4bc3-bfed-f0aa84254b62.xml"
132             ttimsgs = lines.filter(l => l.folder.startsWith('trace')).map(l => l.file.split('_')).filter(t => t.length > 6).map(t => {
133                 return { date: t[0], time: t[1], milli: t[2], thread: t[3], categ: t[4], flow: t[5], id: t[6].slice(0, -4) }
134             })
135             return lines.length
136         }))

175 function msgBy5Min() {
176     var ret = {}
177     for (var h = 0; h < 24; h++) {
178         for (var m = 0; m < 60; m += 5) {
179             const min = String(h).padStart(2, '0') + ':' + String(m).padStart(2, '0')
180             ret[min] = 0
181         }
182     }
183     ttimsgs.forEach(m => {
184         var key = m.time.slice(0, 2) + ':' + m.time[2]
185         if (m.time[3] < 5)
186             key += '0'
187         else
188             key += '5'
189         ret[key] = ret[key] + 1
190     })
191     return ret
192 }
